Parse Yaml/JSON file and create loop with keys and values

input_file = open(f"{ks}", "rt")
texts = input_file.read()

with open(f'{var_file}') as f:
      docs = yaml.load_all(f, Loader=yaml.FullLoader)
      for x in docs:
           for k,v in x.items():
                texts = texts.replace(f'${k}', f'{v}')

input_file = open(f"{ks}", "wt")
input_file.write(texts)
input_file.close()
